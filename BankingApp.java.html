import java.util.Scanner;

public class BankingApp {
    static double balance = 10000;


    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int opt = 0;
        while (opt != 6) {
            System.out.println("Welcome to the Java Bank ");
            System.out.println("1. Create an Account");
            System.out.println("2. Check Balance");
            System.out.println("3. Deposit");
            System.out.println("4. Withdraw");
            System.out.println("5. Transfer");
            System.out.println("6. Exit");
            System.out.print("Enter an option: ");
            opt = scanner.nextInt();

            switch (opt) {
                case 1:
                    createacc();
                    break;
                case 2:
                    checkBalance();
                    break;
                case 3:
                    deposit();
                    break;
                case 4:
                    withdraw();
                    break;
                case 5:
                    transfer();
                    break;
                case 6:
                    exit();
                    break;
                default:
                    System.out.println("Invalid option. Try again.");
                    break;

            }return;
        }
    }

    public static void checkBalance() {
        System.out.println("Your current balance is $" + balance);
        return;
    }

         Scanner scanner = new Scanner(System.in);
         double balance=0.0;

    }

    public static void deposit() {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter the amount to deposit: $");
        double amount = scanner.nextDouble();
        balance += amount;
        System.out.println("$" + amount + " has been deposited to your account.");
        checkBalance();
    }

    public static void withdraw() {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter the amount to withdraw: $");
        double amount = scanner.nextDouble();
        if (amount > balance) {
            System.out.println("Insufficient funds.");
        } else {
            balance -= amount;
            System.out.println("$" + amount + " has been withdrawn from your account.");
        }
        checkBalance();
    }

    public static void transfer() {
        Scanner scanner = new Scanner(System.in);
        double minamt = 500;
        System.out.print("Enter the source Account Number");
        int srcacc = scanner.nextInt();
        System.out.print("Enter the Destination Account Number");
        int destacc = scanner.nextInt();
        System.out.print("Enter the amount to transfer: $");
        double transamt = scanner.nextDouble();
        if (transamt >= minamt && transamt <= balance) {
            double totbal = balance - transamt;
            System.out.println("Transfered Amount=" + transamt);
            System.out.println("Your current balance is $" + totbal);
        } else {
            System.out.println("Insufficient balance");
        }

    }

    public static void exit() {
        System.out.println("Thank you for banking with us. Have a great day!");
    }

}
